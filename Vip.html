// Función para calcular promedio de una semana - VERSIÓN SIMPLIFICADA function calcularSemana(semana) { let suma = 0; // Sumar todos los días de la semana (siempre 6 días) for(let i=1; i<=6; i++) { let input = document.getElementById(semana + 'd' + i); let valor = parseFloat(input.value) || 0; // Si no es número válido, usar 0 suma += valor; } // Calcular promedio (dividir entre 6) let promedio = suma / 6; let promedioFixed = promedio.toFixed(1); // Actualizar el total de la semana document.getElementById(semana + 'Total').innerText = promedioFixed + '%'; return parseFloat(promedioFixed); }
